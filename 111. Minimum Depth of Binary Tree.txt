/** * Definition for a binary tree node. * struct TreeNode { *     int val; *     TreeNode *left; *     TreeNode *right; *     TreeNode(int x) : val(x), left(NULL), right(NULL) {} * }; */class Solution {public:    int minDepth(TreeNode* root) {        if (root == NULL)            return 0;                bool l,r;        int s;        queue<pair<TreeNode*,int>> q;        q.push(make_pair(root, 1));        q.push(make_pair(root, 1));                do{            q.pop();            root = q.front().first;            s = q.front().second;            r = l = false;                        if (root->left != NULL){                l = true;                q.push(make_pair(root->left, s+1));            }                        if (root->right != NULL){                r = true;                q.push(make_pair(root->right, s+1));            }        }while(l || r);                return s;    }};
