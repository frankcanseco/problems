class Solution {public:    vector<int> v;    void merge(vector<int>& nums1, int m, vector<int>& nums2, int n) {        int s = n + m - 1;        m--;        n--;        for (int i = s; i >= 0 ; i--){            if (n < 0 || m < 0)                break;                        if (nums1[m] > nums2[n]){                nums1[i] = nums1[m--];            } else{                nums1[i] = nums2[n--];            }        }                for (int i = n; i >= 0 ; i--)            nums1[i] = nums2[i];    }};
